Terminal 
# Clone e setup automático
git clone https://github.com/MrJc01/crompressor-projetos
cd crompressor-projetos
chmod +x scripts/*.sh
./scripts/setup.sh

# Abrir Hub de Simuladores
./scripts/run_all_sims.sh
# → http://localhost:8091/hub/
